page_count = 1 #init page count to keep track
link_list = [] #init list of links
def get_links(url, page = 1):
    '''
    This function takes a base url for the category of new and fetches the links for the desired 
    number of articles given by the page argument.
    Input: string, base url
    Output: list of article url strings.
    '''
    #declare global vars.
    global res_links 
    global page_count
    global link_list
    #special base url if just 1 page requested
    if page_count == 1:
        req = requests.get(url) #request from url
    #other amount of pages
    else: req = requests.get('{}page/{}/'.format(url, page_count))
    #continue if request successful
    if req.status_code == 200:
        #use beautifulsoup for extracting link
        bsoup_obj = BeautifulSoup(req.text, 'lxml') 
        for e in bsoup_obj('h2'): #shortcut for bsou_obj.find_all
            link_list.append(str(e.a['href'])) #append found link
    else: #for unsuccessful request
        page_count = page #set page_count to cease 
    page_count += 1 #count page
    if page_count <= page: #self-call function if not enough pages yet
        get_links(url, page) #call self
    else: #copy into another list to return since link_list reset with each self-call
        res_links = list(link_list)
        link_list = [] #avoid duplicates
        page_count = 1 #reset page_count
    return res_links

def get_info(url):
    '''
    This function uses the provided article url to grab all the necessary information.
    Input: string, article url
    Output: dictionary, url, title, text, author
    '''
    info_dict = {} #init dict
    req = requests.get(url) #request
    soup = BeautifulSoup(req.text, 'lxml')
    
    article = soup.find('article') #locate the article tag portion
    title = article.h1.string #all titles in h1 tag 
    #check for newline in last paragraph tag
    #if not, then author is in last para. of its own
    if '\n' not in article('p')[-1].contents:
        author = article('p')[-1].get_text() #author in last para.
        if 'Written' not in author: #check exception
            author = article('p')[-2].get_text() #try 2nd to last para
        #body text of article- get all text from p tags if not in author
        text = (''.join([e.get_text() for e in article('p')[:] if 
        e.get_text() not in author]))
    else: #exeption-author is in last p. tag with with some body text
        #get author by splitting by newline and grabbing last portion
        author = article('p')[-1].get_text().split('\n')[-1]
        #text is portion before newline
        text = ''.join([e.get_text() for e in article('p')[:-1]]).split('\n')[0]
    if "Written" not in author: #some article has no author line (police log)
        author = 'No author'
    #translate some unicode chars
    text = text.translate({ 0x2018:0x27, 0x2019:0x27, 0x201C:0x22, 0x201D:0x22, 0x2026:0x20 })
    #final dict of all request info.
    info_dict = {'url': url, 'title': title, 'text': text, 'author': author}
    return info_dict

def build_df(base_url, entries):
    '''
    This function takes a base url for news category as well as number of articles and returns 
    a pd dataframe of all their respective info. Also add a column for category.
    Input: string, int, base url and number of entries
    '''
    #use function get_link, entries/15 for # of pages
    links = get_links(base_url, entries / 15)
    #init a dictionary with desired keys, values are lists
    all_info = {'author':[],'text':[],'title':[],'url':[]}
    for e in links: #get info for each link 
        info = get_info(e) #call get_info 
        for key in info.iterkeys(): #append to list of values for each key
            all_info[key].append(info[key])
    df = pd.DataFrame(all_info) #turn into pd dataframe
    #add category column 
    df['category'] = base_url.split('/')[-2] + ' news' 
    return df

    return [lemma(tok.lower()) for tok in tokenize(text)]

def top_words(corpus, word_num = 50):
    '''
    This function takes in the corpus, and # of words desired and returns the top words within that corpus.
    Input: list of strings, each element a body of text- corpus
    Output: list of top words
    '''
    #use tfidf and fit to corpus
    vectorizer = TfidfVectorizer(tokenizer=wnl_lemmatizer,stop_words=eng_stop_mod,smooth_idf=True,norm=None)
    fit = vectorizer.fit_transform(corpus)
    farray = fit.toarray() #represent with np array
    farray = np.vstack([farray, sum(farray)]) #additional row for sum of each col.
    #words and vocab (dict of words as keys and index as values)
    word = vectorizer.get_feature_names()
    vocab = vectorizer.vocabulary_
    #invert the vocab dict, keys-ids, values-words
    vocab_rev = {vocab[key]: key for key in vocab} 
    #order the word ids descending by flipping the array up down
    ordered_word_id = list(np.flipud(np.argsort(farray[-1])))
    #list compr.  for list of top words
    top_words = [[vocab_rev[key],farray[-1,key]] for key in ordered_word_id[:word_num]]
    return top_words

def text_comp(text1, text2):
    '''
    simple text comparison, returns words in common
    '''
    fit1 = [w for w in wnl_lemmatizer(text1) if w not in eng_stop_mod]
    fit2 = [w for w in wnl_lemmatizer(text2) if w not in eng_stop_mod]
    return [w for w in fit1 if w in fit2]

According to the fit using tfidf, we can gather some information about the weighted commonly occuring words in the aggie corpus. Basing off of these top words, we can see that many of them can be classified as possible topic words. As expected, among the top results, we observe topics like uc davis, university, student, city, community, food, event, asucd (associated students, uc davis). Within the top 100 words, we also see less locally-focused issues such as climate, science, pipeline, california. Also within the top 100, concurrently relevant topics like milo yiannopoulos, Trump, muslim are also present.

Using the similarity matrix, we examine that on average, the simalarity score between the campus and city articles is 410. Compared to the overall average similarity of 587, the two types of news exhibits about 70% similarity of overall. Typically, we should see overlaps in the topics of city and campus news considering also that they both tend to cover local, university-related news.

The titles of the top 3 pairs of most similar articles are: (UC Davis holds first mental health conference, UC Davis to host first ever mental health conference), ASUCD President Alex Lee vetoes amendment for creation of Judicial Council, Last week in Senate), (2017 ASUCD Winter Elections Meet the Candidates, Student regent recruitment for the 2018-2019 school year begins) The top pair is especially obvious for their similarity in content. Most notable words in common for the first pair include: administration, counseling, conference, community, mental, psychology, and health. The second pair: asucd, voting, president, student, council, court, commission, and, committee. The third pair: candidate, community, affordability, budget, campus, policy, and program. These words in common can be seen in the wordcloud for each pair.

It may be reasonable to think this corpus is representative of the aggie to some degree considering that the other categories of news may be even more locally/ucd cetered. Some examples include sports, arts and culture, and features which likely covers mostly local stories. Having so many topics covering local affairs and sentiments, this corpus could support inferences regarding the on-going topics within and surrounding the community with the occassional coverage of hot-buttoned national issues. Overal, this should be expected of a news group within a university.
